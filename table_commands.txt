CREATE TABLE booking(
key NUMBER(2)CONSTRAINT PK_b100 PRIMARY KEY,
DFROM VARCHAR2(14),
DTO VARCHAR2(14),
RFROM VARCHAR2(14),
RtO VARCHAR2(14))


CREATE TABLE BOOK_FLIGHT(
PASSPORT_NUMBER  VARCHAR2(15)CONSTRAINT PK_b10 PRIMARY KEY,
DFROM VARCHAR2(15),
DTO VARCHAR2(15),
RFROM VARCHAR2(15),
RtO VARCHAR2(15)
DEPARTURE_DATE  DATE,
RETURN_DATE  DATE.,
TOTAL_PRICE NUMBER(15))


CREATE TABLE DEPARTURE(
key NUMBER(2) CONSTRAINT FR_2 REFERENCES BOOKING (key),
FLIGHT_NO VARCHAR2(14),
FLIGHT_DATE  DATE
TIME VARCHAR2(14),
PRICE NUMBER(20))






CREATE TABLE RETURN(
key NUMBER(2) CONSTRAINT FR_3 REFERENCES BOOKING (key),,
FLIGHT_NO VARCHAR2(14),
FLIGHT_DATE  DATE
TIME VARCHAR2(14),
PRICE NUMBER(20))



CREATE TABLE INFORMATION(
PASSPORT_NUMBER  VARCHAR2(15) CONSTRAINT FR_5 REFERENCES BOOK_FLIGHT (PASSPORT_NUMBER ),,,
NAME VARCHAR2(40),
EMAIL VARCHAR2(40),
GENDER  VARCHAR2(14),
BIRTH_DATE DATE,
CARD_NUMBER NUMBER(20),
CARD_TYPE VARCHAR(14).,
CARDHOLDER_NAME VARCHAR2(40),
EXPIRY_DATE_Y  NUMBER(13),
EXPIRY_DATE_M  NUMBER(13),
SECURITY_CODE  NUMBER(3))


CREATE TABLE SEAT(
PASSPORT_NUMBER  VARCHAR2(15) CONSTRAINT FR_6 REFERENCES BOOK_FLIGHT (PASSPORT_NUMBER ),
SEAT  VARCHAR2(15))


CREATE TABLE USERS(
USER_NAME  VARCHAR2(15),
PASSWORD  VARCHAR2(20))
